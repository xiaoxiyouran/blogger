AES 加密和解密

示例

抄的网上一个加密和解密的例子：

#!/usr/bin/env python
# -*- coding: utf-8 -*-

'''
filename: crpyto.py
message:  AES 加密和解密
refer: https://www.cnblogs.com/kaituorensheng/p/4501128.html
https://www.jianshu.com/p/2d0f75a6a074
'''

__author__ = 'xi'

import sys
from Crypto.Cipher import AES
from binascii import b2a_hex, a2b_hex


class prpcrypt():
	def __init__(self, key):
		self.key = key
		self.mode = AES.MODE_CBC

	# 加密函数，如果text不是16的倍数【加密文本text必须为16的倍数！】，那就补足为16的倍数
	def encrypt(self, text):
		cryptor = AES.new(self.key, self.mode, self.key)
		# 这里密钥key 长度必须为16（AES-128）、24（AES-192）、或32（AES-256）Bytes 长度.目前AES-128足够用
		length = 16
		count = len(text)
		add = length - (count % length)
		text = text + ('\0' * add)
		self.ciphertext = cryptor.encrypt(text)

		# 因为AES加密时候得到的字符串不一定是ascii字符集的，输出到终端或者保存时候可能存在问题
		# 所以这里统一把加密后的字符串转化为16进制字符串
		# print self.ciphertext # 会存在乱码问题
		return b2a_hex(self.ciphertext)

	# 解密后，去掉补足的空格用strip() 去掉
	def decrypt(self, text):
		cryptor = AES.new(self.key, self.mode, self.key)
		plain_text = cryptor.decrypt(a2b_hex(text))
		return plain_text.rstrip('\0')


if __name__ == '__main__':
	pwd = 'keyskeyaaaaa'
	if (len(pwd) < 16):
		pwd = pwd + (16 - len(pwd)) * 'a'
	pc = prpcrypt(pwd)  # 初始化密钥
	e = pc.encrypt("好的")
	d = pc.decrypt(e)
	print e, d
	e = pc.encrypt("你还好吗")
	d = pc.decrypt(e)
	print e, d


输出结果如下：

3907e441d1aae16593ba4bbdbea94dc1 你还好吗
