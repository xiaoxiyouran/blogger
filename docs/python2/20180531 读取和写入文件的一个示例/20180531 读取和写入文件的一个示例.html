20180531 读取和写入文件的一个示例

1- 写出了用行缓冲读取文件最快的方式；

2- 格式化字符串用 % 连接

#!/usr/bin/python
# -*- coding: UTF-8 -*-

def readFromFile(filename):
	res = []
	file = open(filename)

	while 1:
		lines = file.readlines(100000)
		if not lines:
			break
		for line in lines:
			tmpLine = line.split(',', 1);
			res.append(tmpLine[0])


	return res


dataA = readFromFile("./gemm_AUTO/matrixA.dat")
dataB = readFromFile("./gemm_AUTO/matrixB.dat")

f = open("initMemory",  "w")
print  dataA
for i in range(128):
	indexA = i;
	indexDest = i
	if i == 0:
		f.write("@%d \n" % indexDest)

	f.write("%s \n" % dataA[indexA])

for i in range(256):
	indexB = i;
	indexDest = 512 + i
	if i == 0:
		f.write("@%d \n" % indexDest)

	f.write("%s \n" %  dataB[indexB])

for i in range(136):
	indexDest = 1024
	if i == 0:
		f.write("@%d \n" % indexDest)

	f.write("0 \n" )

f.close()
