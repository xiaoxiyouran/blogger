我记得没做完，这是其中一道，求时针和分针之间夹角的一道题，要求是求钝角。

输出要求：如果是整数，直接输出整数，

如果是小数，输出一位小数

#include <bits/stdc++.h>
using namespace std;

int main(){

	string in;
	cin >> in;
	vector<string> v ;
	int pos = -1;
	for(int i=0; i< in.size();i++){
		if(in[i] == ':'){
			pos = i;
			break;
		}
	}

	string hh = in.substr(0,pos-0);
	string mm = in.substr(pos+1,in.size()-1-pos);
//    cout << hh << mm;
	int h = stoi(hh);
	int m = stoi(mm);

	h = (h>12)?(h-12):h;
	float angleH = h * 30 + m/2.0;
	float angleM = m * 6;
	float diff = abs(angleM - angleH);
	if( diff >180 ){
		diff = 360 - diff;
	}

	float t = diff - floor(diff);
	if( t > 1e-6  ){
		printf("%.1f\n",diff);
	}else{
		printf("%.0f\n", diff);
	}

	return 0;
}

这个题是要求输出满足要求k的最大值，最小值？

#include <bits/stdc++.h>
using namespace std;

void mapf(int low ,int high, map<int,int>& mymap){
	for( int i=low; i<=high; i++ ){
		mymap[i] = mymap[i] + 1;
	}

}

int main(){

	map<int,int> mymap;
	int n,k;
	cin >> n>> k;
	for( int i =0; i< n; i++ ){
		int low,  high;
		cin >> low >> high;
		mapf(low, high, mymap);

	}

	int cMin = -100;
	int cMax = -100;
	for( auto it = mymap.begin(); it != mymap.end(); it++ ){
//        cout << it ->first << it->second;
		if(it -> second >= k){
			cMin =  it -> first;
			break;
		}
	}

	for( auto it = mymap.rbegin(); it != mymap.rend(); it++ ){
//        cout << it ->first << it->second;
		if(it -> second >= k){
			cMax =  it -> first;
			break;
		}
	}

	if( cMin == -100 && cMax == -100 )
		cout << "error"<< endl;
	else{
		cout << cMin << " " << cMax << endl;
	}
	return 0;
}
